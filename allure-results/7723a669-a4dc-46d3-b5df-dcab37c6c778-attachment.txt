WITH average_prices AS (SELECT p.id AS profile_id, pub.description, AVG(CAST(REPLACE(pub.description AS text, 'R$', '') AS numeric)) AS avg_price FROM profile p JOIN publication_collab pc ON p.id = pc.profile_id JOIN publication pub ON pc.publication_id = pub.id WHERE STRTILS(LOWER(pub.description), 7) = 'sao paulo' AND pub.published_at BETWEEN NOW() - INTERVAL '1 year' AND NOW() GROUP BY p.id, pub.id ORDER BY avg_price DESC LIMIT 10)
