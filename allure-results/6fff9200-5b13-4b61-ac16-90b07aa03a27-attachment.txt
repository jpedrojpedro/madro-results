WITH publication_words AS (SELECT unnest(regexp_split_to_table(p.description_lexemes, E'\W+')) word FROM publication p), profile_topics AS (SELECT p.profile_id, string_agg(DISTINCT word ORDER BY ts_rank(publication_words.word, to_tsquery('english', :query))) AS topic_string FROM publication_words JOIN publication p ON string_agg(DISTINCT word, ' ') = p.description_lexemes GROUP BY p.profile_id) SELECT profile_id, topic_string FROM profile_topics ORDER BY topic_string DESC LIMIT 10
