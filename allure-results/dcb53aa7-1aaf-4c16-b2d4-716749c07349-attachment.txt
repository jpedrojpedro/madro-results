WITH followed_bakeries AS (SELECT pr.origin_profile_id AS profile_id, p.username FROM publication_collab pc JOIN profile_relationship pr ON pc.publication_id = pr.destination_profile_id AND pr.edge  = 'follows' JOIN public.profile p ON pr.origin_profile_id = p.id WHERE p.is_private = FALSE), bakeries AS (SELECT p.id AS profile_id, p.username FROM followed_bakeries fb LEFT JOIN publication_collab ON fb.profile_id = publication_collab.profile_id AND id IN (SELECT publication_id FROM publication_collab) WHERE biography_lexemes @@ to_tsquery('english', '(baker* OR bakery)')), most_popular_first AS (SELECT * FROM bakeries ORDER BY num_followers DESC), result_set AS ( SELECT p.id AS profile_id, p.username FROM public.profile p JOIN most_popular_first mpf ON p.id = mpf.profile_id LIMIT 10) SELECT p.id AS profile_id, p.username FROM result_set rs JOIN most_popular_first mpf ON rs.profile_id = mpf.profile_id ORDER BY ts_rank(mpf.biography_lexemes, to_tsquery('english', '(baker* OR bakery)')) DESC LIMIT 10;
