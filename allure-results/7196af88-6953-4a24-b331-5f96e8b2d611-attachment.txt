SELECT pc.publication_id, pr.origin_profile_id AS profile_id FROM public.profile_relationship pr JOIN publication_collab pc ON pr.destination_profile_id = pc.profile_id LEFT JOIN public.profile p ON pc.profile_id = p.id WHERE p.num_followers > 10000 AND (p.username ILIKE 's%paule%' OR p.full_name ILIKE '%Sao Paulo' OR (p.biography LIKE '%restaurant%' OR p.biography_lexemes @@ to_tsquery('english', '' || quote_literal('Sao Paulo') || ''))) ORDER BY CASE WHEN pc.publication_id IS NULL THEN ts_rank(p.biography_lexemes, to_tsquery('english', '' || quote_literal('Sao Paulo') || '')) ELSE ts_rank(pc.profile.collaborators_biography_lexemes, to_tsquery('english', 'restaurant')) END DESC LIMIT 10
