WITH candidates AS (SELECT pub.id AS profile_id FROM publication pub JOIN profile p ON pub.profile_id = p.id WHERE pub.type != 'GraphImage' LIMIT 10) SELECT c.publication_id AS publication_id, ts_rank(pub.description_lexemes, to_tsquery('english', :request)) * 100 AS rank FROM comment c FULL OUTER JOIN candidates cands ON c.publication_id = cands.profile_id INNER JOIN publication pub ON c.publication_id = pub.id WHERE pub.published_at BETWEEN CURRENT_DATE - INTERVAL '1 month' AND CURRENT_DATE ORDER BY ts_rank(pub.description_lexemes, to_tsquery('english', :request)) DESC LIMIT 10
