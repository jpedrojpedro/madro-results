WITH publication_comment_ids AS ( SELECT b.id, COUNT(c.id) AS comment_count FROM publication p JOIN comment c ON p.id = c.publication_id WHERE lower(c.annotation) LIKE '%' || 'steakhouse' || '{}' OR lower(p.description) LIKE '%' || 'steakhouse' || '{}' GROUP BY b.id ), engagement_scores AS ( SELECT p.id as profile_id, bc.num_comments + bc.num_likes / (bc.num_followers + 1) AS engagement_score FROM profile b JOIN publication p ON b.id = p.profile_id LEFT JOIN comment c ON p.id = c.publication_id LEFT JOIN ( SELECT publication_id, COUNT(id) AS num_comments, SUM(num_likes) / NULLIF(SUM(num_followers), 0) AS engagement_score FROM comment GROUP BY publication_id ) bc ON c.publication_id = bc.publication_id WHERE c.annotation ILIKE '%steakhouse%' ORDER BY engagement_score DESC LIMIT 10 ), rankedPublicationIds AS ( SELECT p.id AS profile_id, ts_rank(p.description_lexemes, to_tsquery('english', 'steakhouses*')) + ts_rank(b.biography_lexemes, to_tsquery('english', 'steakhouse')) OVER (PARTITION BY b.id) AS relevance_score FROM publication_comment_ids pci JOIN engagement_scores es ON pci.profile_id = es.profile_id JOIN publication p ON pci.profile_id = p.id JOIN profile b ON pci.profile_id = b.id ORDER BY relevance_score DESC LIMIT 10 ) SELECT * FROM rankedPublicationIds
