SELECT p.id AS profile_id FROM publication_collab pc JOIN profile p ON pc.profile_id = p.id WHERE (pc.publication_id IN ( SELECT id FROM publication WHERE description @@ to_tsquery('english', 'fluminense fc'::tsq) LIMIT 1 ) OR pc.profile_id = ( SELECT origin_profile_id FROM profile_relationship WHERE destination_profile_id IN ( SELECT id FROM profile WHERE username = 'fluminensefc' ) AND edge = 'follows' LIMIT 1 )) ORDER BY pc.visited_at DESC LIMIT 10
