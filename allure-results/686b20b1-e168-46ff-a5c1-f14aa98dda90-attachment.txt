SELECT
  p.id AS profile_id,
  p.username,
  p.full_name,
  ts_rank(p.biography_lexemes, to_tsquery('english', 'health:* & (restaurant:* | nutrition:* | food:*)')) AS rank
FROM profile p
WHERE p.biography_lexemes @@ to_tsquery('english', 'health:* & (restaurant:* | nutrition:* | food:*)')
  AND (p.is_brand = TRUE OR p.full_name ILIKE '%restaurant%' OR p.biography ILIKE '%restaurant%')
ORDER BY rank DESC, p.num_followers DESC
LIMIT 10
