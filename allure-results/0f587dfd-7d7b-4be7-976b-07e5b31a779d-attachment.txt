SELECT p.id AS profile_id, ts_rank(p.biography_lexemes, to_tsquery('english', 'bakery')) FROM ( SELECT DISTINCT pc.profile_id AS profile_id FROM publication_collab AS pc INNER JOIN publication AS p ON pc.publication_id = p.id WHERE NOT is_private AND p.bio_graphy LIKE '%bakery%' ) AS influencers_publications , `profile` AS p WHERE p.id IN ( SELECT profile_id::integer FROM influencers_publications ) ORDER BY ts_rank(p.biography_lexemes, to_tsquery('english', 'bakery')) DESC LIMIT 10
