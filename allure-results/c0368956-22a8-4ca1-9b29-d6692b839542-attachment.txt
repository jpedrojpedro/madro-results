WITH pubs AS (SELECT p.id AS publication_id, ts_rank(p.description_lexemes, 'english'::regconfig, to_tsquery('english','Tutto Nhoque')) AS relevance FROM publication p ) , profs_with_biography AS ( SELECT r.publication_id as profile_id, COALESCE(pr.destination_profile_id, pr.origin_profile_id) AS restaurant_id, p.relevance FROM pubs p JOIN profile_relationship pr ON p.publication_id = pr.destination_profile_id ), restaurants AS ( SELECT profs_with_biography.restaurant_id , profs_with_biography.profile_id AS profile_id, ts_rank(t.biography_lexemes, 'english'::regconfig, to_tsquery('english','Tutto Nhoque')) * 1000 AS username_relevance FROM pubs p JOIN publication_collab pc ON p.publication_id = pc.publication_id AND (pc.collaborators = true OR pc.collaboration_id IS NULL OR pc.parent_publication_id IS NULL) JOIN publiaiton p2 ON p2.id = profs_with_biography.restaurant_id GROUP BY profs_with_biography.restaurant_id ORDER BY username_relevance DESC LIMIT 10
