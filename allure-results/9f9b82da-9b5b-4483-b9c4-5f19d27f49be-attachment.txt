WITH profiles AS ( SELECT p.id AS profile_id, ts_rank(p.biography_lexemes, to_tsquery('italian', 'pancetta') + ts_rank(p.biography_lexemes, to_tsquery('italian', 'prosciutto'))) AS rank FROM public.profile p WHERE biography_lexemes @@ to_tsquery('english', 'italian pancetta prosciutto') AND (p.biography LIKE '%pancetta%' OR p.biography LIKE '%prosciutto%') ORDER BY rank DESC LIMIT 10 )
