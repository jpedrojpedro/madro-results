WITH viral_profile AS (SELECT p.id AS profile_id, COUNT(c.publication_id) AS comments_count, SUM(p.num_followers + c.num_likes + c.num_comments)  AS engagement_score FROM public.profile AS p LEFT JOIN public.comment AS c ON p.id = c.profile_id WHERE EXTRACT(MONTH FROM CURRENT_DATE + INTERVAL '1 month') * 100 + EXTRACT(DAY FROM CURRENT_DATE + INTERVAL '1 month') BETWEEN p.published_at AND (CURRENT_TIMESTAMP::date + INTERVAL '1 day') GROUP BY p.id ORDER BY ts_rank(p.biography_lexemes, to_tsquery('english', ''viral restaurant'')) DESC LIMIT 10)
