WITH target_profile AS (
    SELECT id 
    FROM profile 
    WHERE username = 'broxadasinistra'
),
target_publications AS (
    SELECT p.id
    FROM publication p
    JOIN target_profile tp ON p.profile_id = tp.id
)
SELECT 
    c.profile_id AS profile_id, 
    c.annotation AS comment_text,
    c.num_likes
FROM comment c
JOIN target_publications tp ON c.publication_id = tp.id
WHERE c.annotation ILIKE '%crítica%' 
   OR c.annotation ILIKE '%pior%' 
   OR c.annotation ILIKE '%errado%'
   OR c.annotation ILIKE '%discordo%'
ORDER BY c.num_likes DESC
LIMIT 10
