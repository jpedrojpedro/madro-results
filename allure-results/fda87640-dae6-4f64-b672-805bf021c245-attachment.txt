WITH discussion_profiles AS (SELECT c.profile_id AS profile_id, p.id AS publication_id, ts_rank(p.description_lexemes, to_tsquery('english', '"steakhouses Curitiba"')) AS relevance FROM comment c JOIN publication p ON c.publication_id = p.id WHERE p.description IS NOT NULL AND p.is_private IS FALSE ORDER BY relevance DESC LIMIT 600) SELECT c.profile_id ,p.id AS profile_id ,sum(c.num_comments + (d.relevance * 10000)) as discussion_score FROM public.comment c JOIN discussion_profiles d ON(c.profile_id)=d.profile_id LEFT JOIN publication p ON c.publication_id =p.id GROUP BY c.profile_id ORDER BY COALESCE(sum(c.num_comments),0) DESC, sum(d.relevance) DESC LIMIT 10
