SELECT c.profile_id AS profile_id, ts_rank(to_tsvector(c.annotation), to_tsquery('english', 'haight_clothing')) * 10000 + p.num_followers FROM comment c INNER JOIN publication p ON c.publication_id = p.id WHERE (c.annotation ILIKE '%haight_clothing%') OR (c.publication_id IN (SELECT id FROM publication pp WHERE pp.slug LIKE ('%' || 'haight_clothing' || '%'))) INNER JOIN profile pr ON c.profile_id  = pr.id ORDER BY ts_rank(to_tsvector(c.annotation), to_tsquery('english', 'haight_clothing')) * 10000 + p.num_followers DESC LIMIT 10;
