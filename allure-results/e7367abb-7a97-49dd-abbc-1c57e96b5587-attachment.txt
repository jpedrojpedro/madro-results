SELECT publication_id AS profile_id, p.id, COALESCE(CASE WHEN m.description_lexemes @@ to_tsquery('english', 'healthy restaurant option') THEN ts_rank(m.description_lexemes, to_tsquery('english', 'healthiest restaurant option')) END * 1000 + (m.num_likes > 0 OR m.num_comments > 0), 0) AS engagement FROM publication m JOIN profile p ON m.profile_id = p.id ORDER BY engagement DESC LIMIT 10
