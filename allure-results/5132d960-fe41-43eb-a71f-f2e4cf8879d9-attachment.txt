SELECT p.profile_id, c.annotation AS criticism FROM comment AS c JOIN publication AS p ON c.publication_id = p.id WHERE p.profile_id IN ( SELECT origin_profile_id FROM profile_relationship WHERE destination_profile_id = ( SELECT id FROM profile WHERE username = '@broxadasinistra' ) ) AND to_tsvector('english', c.annotation) @@ to_tsquery('english', 'criticism') ORDER BY ts_rank(to_tsvector('english', c.annotation), to_tsquery('english', 'criticism')) DESC LIMIT 10
