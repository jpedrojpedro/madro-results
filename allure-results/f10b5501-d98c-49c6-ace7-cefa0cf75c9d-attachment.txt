WITH brand_profiles AS (SELECT id FROM public.profile WHERE lower(username) LIKE 'haight_clothing%') SELECT pr.destination_profile_id AS profile_id, p.id AS publication_id, pr.edge FROM public.publication_collab pc JOIN brand_profiles b ON pc.profile_id = b.id JOIN public.publication p ON pc.publication_id = p.id JOIN public.profile_relationship pr ON (pr.origin_profile_id = p.id AND pr.destination_profile_id = b.id) WHERE pr EDGE='follows' ORDER BY published_at DESC LIMIT 10;
