SELECT c3.profile_id, ts_rank(c2.description_lexemes, to_tsquery('english', 'commenting on ' || p1.description)) AS engagement FROM public.publication p1 INNER JOIN publication_collab pc6 ON (p1.id = pc6.publication_id) INNER JOIN profile p2 ON (pc6.profile_id=p2.id) LEFT JOIN public.comment c3 ON ((c3.publication_id=pc6.publication_id ) OR (c3.reply_to IS NULL)) WHERE pc6.profile_id = $1 ORDER BY engagement DESC LIMIT 10;
