SELECT p.profile_id, ts_rank(p.description_lexemes, to_tsquery('english', 'crisis complaint')) AS relevance_score FROM publication AS p JOIN profile_relationship AS r ON p.profile_id = r.origin_profile_id AND r.edge = 'follows' WHERE r.destination_profile_id IN (SELECT id FROM profile WHERE username = '@fluminensefc') AND published_at > NOW() - INTERVAL '1 month' ORDER BY relevance_score DESC LIMIT 10
