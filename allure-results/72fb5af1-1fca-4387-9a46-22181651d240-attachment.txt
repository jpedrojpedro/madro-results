SELECT 
    p.id AS profile_id, 
    p.username, 
    p.biography
FROM profile p
WHERE p.biography_lexemes @@ to_tsquery('english', 'nutrition | nutritional | breakdown | calories | protein | fat | carbs')
ORDER BY ts_rank(p.biography_lexemes, to_tsquery('english', 'nutrition | nutritional | breakdown | calories | protein | fat | carbs')) DESC
LIMIT 10;
