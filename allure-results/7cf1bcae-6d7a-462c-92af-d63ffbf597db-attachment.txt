SELECT
  p.id AS profile_id,
  p.username,
  p.full_name,
  ts_rank(p.biography_lexemes, websearch_to_tsquery('portuguese', 'Balneário Camboriú restaurante')) AS rank
FROM profile p
WHERE p.biography_lexemes @@ websearch_to_tsquery('portuguese', 'Balneário Camboriú restaurante')
ORDER BY rank DESC
LIMIT 10;
