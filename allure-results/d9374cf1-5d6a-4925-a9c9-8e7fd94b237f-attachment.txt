SELECT 
    p.id AS profile_id, 
    p.username, 
    p.full_name, 
    p.biography,
    p.num_followers
FROM profile p
WHERE p.biography_lexemes @@ to_tsquery('english', 'restaurant & (owner | chef | cook | host)')
ORDER BY ts_rank(p.biography_lexemes, to_tsquery('english', 'restaurant & (owner | chef | cook | host)')) DESC, p.num_followers DESC
LIMIT 10;
