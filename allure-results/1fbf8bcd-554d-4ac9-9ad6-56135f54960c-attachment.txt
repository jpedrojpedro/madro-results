WITH personality_query AS ( SELECT p.id AS profile_id, ts_rank(biography_lexemes, to_tsquery('english', 'best')) * ts_rank(biography_lexemes, to_tsquery('english', 'personality')) AS relevance FROM public.profile WHERE biography IS NOT NULL ORDER BY relevance DESC LIMIT 10 ) SELECT p.id AS publication_id, pr.origin_profile_id AS profile_id from personality_query, public.publication left join public.profile_relationship as pr on (publication.publication_id = pr.publication_id) ;
