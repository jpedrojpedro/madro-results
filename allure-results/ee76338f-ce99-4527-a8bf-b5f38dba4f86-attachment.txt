SELECT
  p.id AS publication_id,
  p.description,
  p.published_at
FROM publication p
WHERE
  p.published_at >= '2025-01-01 00:00:00'
  AND p.published_at <= '2025-12-31 23:59:59'
  AND (
    p.description_lexemes @@ to_tsquery('english', 'daily & special')
    OR p.description ILIKE '%daily special%'
  )
ORDER BY p.published_at DESC
LIMIT 10
