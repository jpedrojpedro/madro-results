WITH flagged_comments AS ( SELECT c.profile_id AS profile_id, p.id AS publication_id, ts_rank(biography_lexemes, to_tsquery('english', 'racial slur')) > 0 OR EXISTS( SELECT f FROM flagged_comments f WHERE p.id=f.publication_id AND f.profile_id!=c.profile_id ) ), THEN TRUE ELSE FALSE END AS  is_flagged , published_at , (coalesce(num_likes+1.0*(count(*) OVER (PARTITION BY c.publication_id ORDER BY published_at)), 0) / coalesce( num_comments + count(*),0)) >= 2 OR EXISTS( SELECT f.profile_id FROM flagged_comments f WHERE p.id=f.publication_id AND f.profile_id!=c.profile_id ) THEN TRUE ELSE FALSE END AS is_flagged , c.annotation, num_likes AS profile_total_kudos, num_comments AS total_comments FROM comment c JOIN publication p ON (p.published_at < NOW()::timestamp  - INTERVAL '14d') AND p.profile_id IN (SELECT destination_profile_id  FROM profile_relationship  WHERE origin_profile_id=123456789 AND edge='follows' ) JOIN profile USING(profile_id) WHERE published_at < NOW()::timestamp - INTERVAL '14d' ORDER BY CASE WHEN ts_rank(description_lexemes, to_tsquery('english', 'racial slur')) > 0 THEN 1 ELSE NULL END DESC, coalesce(num_likes+1.0*(count(*) OVER (PARTITION BY c.publication_id ORDER BY published_at)), 0) / coalesce( num_comments + count(*),0 ) DESC LIMIT 10;
