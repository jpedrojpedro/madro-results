WITH profiles AS ( SELECT pp.id AS profile_id, ts_rank(COALESCE(pp.description_lexemes, pp1.biography_lexemes), to_tsquery('english', 'fitness')) as ts_rank FROM public.profile pp JOIN public.publication pp2 ON pp.id = pp2.profile_id WHERE COALESCE(pp.description_lexemes, pp1.biography_lexemes) @@ to_tsquery('english', 'fitness') ), collabs AS ( SELECT pp1.id AS publication_id, COUNT(*) OVER (PARTITION BY pp1.id) > 0 FROM profiles pp1 INNER JOIN public.publication_collab ON pp1.profile_id = profile_id WHERE edge = ANY(ARRAY['collaborate'])) SELECT DISTINCT t2.profile_id AS profile_id, pp.full_name FROM profiles t3 CROSS JOIN collabs c1 LEFT JOIN profiles t2 ON t2.ts_rank > 100 AND (t2.id <> t3.id OR c1.publication_id IS NOT NULL) ORDER BY (c1.publication_id IS NOT NULL), ts_rank, COALESCE(pp.biography_lexemes, pp.description_lexemes) @@ to_tsquery('english', 'fitness') DESC LIMIT 10;
