WITH healthy_options AS (SELECT p.id AS profile_id, publication.id AS publication_id, publication.medicinal_info AS num_medicinalsAvg FROM publication JOIN publication_collab pc ON publication.id = pc.publication_id JOIN profile p ON pc.profile_id = p.id WHERE (p.num_medicinals IS NOT NULL) GROUP BY p.id),  healthy_restaurants AS (SELECT d.p.id AS profile_id, MAX(p.medicinal_info) AS num_proteins_avg FROM healthy_options h JOIN publication p ON h.publication_id = p.id GROUP BY d.p.id ), final_result AS (SELECT * FROM healthy_restaurants nr ORDER BY num_proteins_avg DESC LIMIT 10) SELECT profile_id, num_proteins_avg FROM final_result
