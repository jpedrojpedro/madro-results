WITH publication_similarity_score AS (SELECT MAX(ts_rank(pub.description_lexemes, to_tsquery('english', 'Restaurant'))) as max_score FROM publication pub JOIN publication_collab pc ON pub.id = pc.publication_id JOIN profile p ON pc.profile_id = p.id WHERE p.profile_id IN (SELECT id FROM profile WHERE biography LIKE '%Tutto Nhoque%')), final_similarities AS (SELECT p.profile_id, COUNT(DISTINCT pub.id) OVER () as relative_count, 1 / ((max_publication_similarity_score)::float64) * COUNT(*) OVER () FROM publication pub JOIN profile p ON pub.profile_id = p.id WHERE p.biography_lexemes @@ to_tsquery('english', 'restaurant') AND (p.profile_id IN (SELECT id FROM profile WHERE biography LIKE '%Tutto Nhoque%')) ORDER BY 1 + ((max_publication_similarity_score)::float64) * COUNT(publication_id)), most_similar AS (SELECT DISTINCT p.profile_id, ts_rank(p.biography_lexemes, to_tsquery('english', 'restaurant')), 0 FROM final_similarities fs JOIN profile p ON p.id = fs.profile_id LIMIT 10), merged AS (
  SELECT id FROM  -- Select max similar pub ids 
    (WITH pubs WITH score AND weight AS (
      SELECT id, ts_rank(desc Lexemes, to_tsquery('english', 'Restaurant')) AS score
        FROM publication
        WHERE description IS NOT NULL
        ORDER BY score DESC LIMIT 10
        )
      SELECT ID FROM pubs pdw JOIN most_similar rsp ON response.score >.relative_count * (max_publication_similarity_score)::float64
    ) 
       -- Add the ones without any description, based on weight
     UNION ALL
    SELECT id FROM publication WHERE description IS NOT NULL ORDER BY ts_rank(desc Lexemes, to_tsquery('english', 'Restaurant')) DESC LIMIT 10
  )
), ranked_profiles AS (SELECT DISTINCT profile_id FROM merged ), similar_profiles_with_weight AS (
  SELECT profile_id, ts_rank(biography_lexemes, to_tsquery('english', 'restaurant')) AS score, -- Get the weight of each pub profile from the most_similar table
    0.6 * COALESCE(most_similar.weight , CASE WHEN most_similar.score IS NULL THEN 0 ELSE most_similar.score END) AS relatedness_weight FROM publication p JOIN profile pr ON p.profile_id = pr.id WHERE pr.biography LIKE '%Tutto Nhoque%' OR p.description IS NOT NULL
), ranked_results AS (
  SELECT profile_id FROM similar_profiles_with_weights ORDER BY score DESC NULLS FIRST LIMIT 10
)
   -- Select the top 10 results,
sELECT rp.profile_id, rp.biography  FROM ranked_results rr JOIN profile rp ON rp.id = rr.profile_id ORDER BY score DESC
