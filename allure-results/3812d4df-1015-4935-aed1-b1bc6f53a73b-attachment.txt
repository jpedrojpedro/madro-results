SELECT
  p.id AS profile_id,
  p.username,
  p.full_name,
  p.num_followers
FROM profile p
WHERE 
  p.biography_lexemes @@ to_tsquery('portuguese', 'restaurante & italiano & "são paulo"')
  OR p.biography_lexemes @@ to_tsquery('english', 'italian & restaurant & "sao paulo"')
ORDER BY p.num_followers DESC
LIMIT 10;
