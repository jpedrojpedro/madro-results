WITH publication_ids AS ( SELECT id FROM public.publication WHERE ts_rank(description_lexemes, to_tsquery('english', 'italian')) > 0.1 AND (description LIKE '%restaurant%' OR ts_rank(description_lexemes, to_tsquery('english', 'sao paulo')) > 0.1) ), profile_pubs AS ( SELECT p.id AS profile_id FROM public.profile p LEFT JOIN publication_collab pc ON p.id = pc.profile_id WHERE pc.publication_id IN ( SELECT id FROM publication_ids ) AND p.biography like '%sao paulo%' OR ts_rank(p.biography_lexemes, to_tsquery('english', 'italian restaurant')) > 0.1), pubs_to_join AS ( SELECT distinct publication_id FROM public.publication WHERE ts_rank(description_lexemes, to_tsquery('english', 'sao paulo restaurant')) > 0.1 AND (description LIKE '%restaurant%' OR ts_rank(description_lexemes, to_tsquery('english', 'italian')) > 0.1) ), filtered_pubs AS ( SELECT p.id AS profile_id FROM pubs_to_join JOIN LATERAL ( SELECT distinct publication_id FROM public.publication WHERE id = pub_to_join.publication_id ORDER BY ts_rank(description_lexemes, to_tsquery('english', 'sao paulo')) DESC LIMIT 1 ) AS top ON TRUE LEFT JOIN LATERAL (SELECT id FROM public.profile) p ON true WHERE  MATCH(p.biography, 'Italian|Sao Paulo') AGAINST( '', boolean_config = 'bool_and') ), final_selection AS ( SELECT DISTINCT pp.id AS profile_id FROM publication_ids INNER JOIN filtered_pubs pp ON TRUE ORDER BY ts_rank(biography_lexemes, to_tsquery('english', 'italian restaurant')) DESC LIMIT 10;
