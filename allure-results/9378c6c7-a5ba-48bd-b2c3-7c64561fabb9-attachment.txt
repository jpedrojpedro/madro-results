WITH relevant_profiles AS ( SELECT pc.publication_id, p.biography_lexemes FROM publication_collab pc LEFT JOIN profile p ON pc.profile_id = p.id ), query_biography AS ( SELECT id AS profile_id, biography_lexemes FROM public.profile WHERE id = '145231' ) SELECT pp1.publication_id AS publication_id, ts_rank(pp1.description_lexemes, qbi.biography) * count_t.* FROM relevant_profiles pp1 JOIN publication pt ON pp1.publication_id = pt.id CROSS JOIN ( SELECT COUNT(DISTINCT t.profile_id) as count_t, t.id FROM publication_collab pc JOIN public.collaborator c ON pc.profile_id = c.id JOIN publication t ON c.publication_id = t.id GROUP BY t.id ) count_t JOIN query_biography qbi ON pp1.publication_id = qbi.profile_id ORDER BY ts_rank(pp1.description_lexemes, qbi.biography) * count_t DESC LIMIT 10
