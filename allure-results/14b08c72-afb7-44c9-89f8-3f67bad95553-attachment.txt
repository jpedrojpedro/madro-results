WITH target_profile AS (
    SELECT id FROM profile WHERE username = 'eusougabriela'
),
target_publications AS (
    SELECT id FROM publication WHERE profile_id = (SELECT id FROM target_profile)
)
SELECT 
    c.profile_id AS profile_id,
    c.annotation,
    c.num_likes
FROM comment c
WHERE c.publication_id IN (SELECT id FROM target_publications)
  AND (
    c.annotation ILIKE '%crítica%' 
    OR c.annotation ILIKE '%pior%' 
    OR c.annotation ILIKE '%odiei%' 
    OR c.annotation ILIKE '%ruim%' 
    OR c.annotation ILIKE '%horrível%'
  )
ORDER BY c.num_likes DESC
LIMIT 10;
