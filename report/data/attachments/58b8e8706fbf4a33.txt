SELECT p.id AS profile_id FROM publication AS pub JOIN profile AS p ON pub.profile_id = p.id WHERE extract(YEAR FROM current_date + interval '1 month') = extract(YEAR FROM published_at) AND extract(MONTH FROM current_date + interval '1 month') - 1 = extract(MONTH FROM published_at) AND lower(description_lexemes @@ to_tsquery('english', 'restaurant')) ORDER BY ts_rank(description_lexemes, to_tsquery('english', 'restaurant')) DESC LIMIT 10
