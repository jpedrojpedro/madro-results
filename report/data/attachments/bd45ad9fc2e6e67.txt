WITH publication_data AS (SELECT p.id AS profile_id, COALESCE(p.price::numeric, p.price * 5 / COALESCE(n.num_likes, 1)) AS price FROM publication p JOIN LATERAL (SELECT id, num_likes, profile_id FROM profile WHERE is_private = false AND is_brand = false AND is_verified = true AND id IN (SELECT destination_profile_id FROM profile_relationship WHERE edge = 'collab' OR edge = 'mentioned')) n ON p.profile_id = n.profile_id LEFT JOIN profile_relationship r ON p.id = r.destination_profile_id), avg_price AS (SELECT AVG(price) AS average_price FROM publication_data), pub_data_with_avg AS (SELECT pd.profile_id, pd.price, ap.average_price FROM publication p JOIN publication_data pd ON p.id = pd.profile_id LEFT JOIN avg_price ap ON 1 = 1) SELECT * FROM pub_data_with_avg ORDER BY price LIMIT 10;
