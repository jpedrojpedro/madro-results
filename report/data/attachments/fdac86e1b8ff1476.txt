SELECT 
    p.id AS profile_id,
    pub.description AS fitness_tip,
    pub.num_likes
FROM profile p
JOIN publication pub ON p.id = pub.profile_id
WHERE p.username = 'eusougabriela'
  AND (pub.description_lexemes @@ to_tsquery('portuguese', 'treino | exercicio | saudavel | fitness | saude'))
ORDER BY ts_rank(pub.description_lexemes, to_tsquery('portuguese', 'treino | exercicio | saudavel | fitness | saude')) DESC, pub.num_likes DESC
LIMIT 10;
