WITH ranked_publications AS (
 SELECT pub.id AS publication_id, pr.origin_profile_id AS profile_id,
 ts_rank(pub.description_lexemes, 'taste') AS relevance
 FROM publication pc LEFT JOIN publication_relationship pr ON pc.profile_id = pr.destination_profile_id 
 LEFT JOIN publication pub ON pr.origin_profile_id = pub.id WHERE ts_headline(pub.description_lexemes, 'taste', '''best''') > 0 ORDER BY pub.enriched_at DESC LIMIT 10
