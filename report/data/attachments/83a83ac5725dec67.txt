SELECT p.id AS profile_id, p.username, p.full_name, p.biography
FROM profile p
WHERE p.biography_lexemes @@ plainto_tsquery('english', 'healthy restaurant')
   OR p.biography_lexemes @@ plainto_tsquery('english', 'nutrition food')
ORDER BY ts_rank(p.biography_lexemes, plainto_tsquery('english', 'healthy restaurant nutrition')) DESC
LIMIT 10;
