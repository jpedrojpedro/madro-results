SELECT c.profile_id, p.id AS publication_id, ts_rank(publication.description_lexemes, to_tsquery('english', :search_query)) AS relevance FROM public.publicationcollab pc JOIN public.publication p ON pc.publication_id = p.id JOIN public.comment c ON c.publication_id = p.id WHERE (p.username ILIKE ANY(ARRAY[:user_handle] ::text[]) OR (c.annotation ILIKE ANY(ARRAY[:search_query] ::text[]))) ORDER BY ts_rank(publication.description_lexemes, to_tsquery('english', :search_query)) DESC LIMIT 10
