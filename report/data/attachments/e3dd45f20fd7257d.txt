SELECT p.id AS profile_id, pp.id AS publication_id FROM profile p INNER JOIN profile_relationship pr ON p.id = pr.destination_profile_id INNER JOIN publication pp ON pr.origin_profile_id = pp.profile_id WHERE (pp.description_lexemes @@ to_tsquery('english', 'restaurant') OR pp.description_lexemes @@ to_tsquery('english', 'cuisine')) AND (pp.description_lexemes @@ to_tsquery('english', 'fish') OR pp.description_lexemes @@ to_tsquery('english', 'seafood')) LIMIT 10
