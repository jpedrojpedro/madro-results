WITH top_comments AS ( SELECT c.profile_id, COUNT(*) AS count FROM comment c JOIN publication p ON p.id = c.publication_id WHERE to_tsvector('english', c.annotation) @@ to_tsquery('english', 'biggest%criticisms') OR to_tsvector('english', p.description) @@ to_tsquery('english', 'biggest%criticisms') GROUP BY c.profile_id ORDER BY count DESC LIMIT 10 ), profile_data AS ( SELECT p.id AS profile_id, COALESCE(pc.count, 0) AS publication_count, CASE WHEN pc.count > 0 THEN pm.description ELSE NULL END AS most_critical_comment FROM profile p LEFT JOIN LATERAL ( SELECT COUNT(*) AS count FROM comment c JOIN publication pcm ON pcm.id = c.publication_id WHERE to_tsvector('english', c.annotation) @@ to_tsquery('english', 'biggest%criticisms') ) pc ON TRUE LEFT JOIN publication pm ON pm.id IN (SELECT pm.id FROM publication pm JOIN publication_collab pcmcp ON pcmcp.publication_id = pm.id WHERE pcmcp.collaborator_id = p.id AND to_tsvector('english', pm.description) @@ to_tsquery('english', 'biggest%criticisms')) ORDER BY pc.count DESC LIMIT 1 ), combined AS ( SELECT profile_id, publication_count FROM top_comments UNION ALL SELECT * FROM profile_data ) SELECT profile_id, publication_count FROM combined WHERE publication_count <> 0 ORDER BY ts_rank( biography_lexemes, to_tsquery('english', 'biggest%criticisms')) DESC LIMIT 10
