SELECT
  p.id AS profile_id,
  p.username,
  p.full_name,
  p.biography
FROM profile p
WHERE
  p.biography_lexemes @@ to_tsquery('english', 'italian & (restaurant | restaurante) & (rio | rio_de_janeiro)')
ORDER BY
  ts_rank(p.biography_lexemes, to_tsquery('english', 'italian & (restaurant | restaurante) & (rio | rio_de_janeiro)')) DESC
LIMIT 10;
