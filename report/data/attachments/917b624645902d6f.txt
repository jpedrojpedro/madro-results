WITH filtered_publication_ids AS (SELECT DISTINCT pc.publication_id FROM public.publication_collab pc JOIN public.publication p ON pc.publication_id = p.id WHERE p.is_private IS FALSE AND pc.id IS NOT NULL), filtered_ids_for_comment AS (SELECT publication_id, 'publication' AS source_type FROM public.publication WHERE is_private IS FALSE UNION SELECT profile_id, 'profile' AS source_type FROM public.profile WHERE is_private IS FALSE AND id IN ( SELECT origin_profile_id FROM public.profile_relationship pr JOIN filtered_publication_ids fpi ON pr.origin_profile_id = fpi.publication_id ) UNION SELECT profile_id FROM public.publication_collab pc JOIN filtered_publication_ids fpi ON pc.publication_id = fpi.publication_id ), comment_like_scores AS (SELECT c.id, ((c.num_likes + fp.num_followers)::numeric / COALESCE(c.num_comments, 1.0) * COALESCE(fp.num_followers, 1)) as score FROM filtered_ids_for_comment fid ON c.profile_id = fid.publication_id OR c.publication_id = fid.publication_id JOIN public.comment c ON true JOIN public.profile p ON c.profile_id = p.id JOIN filtered_publication_ids fpi ON c.publication_id = fpi.publication_id JOIN public.follow_score_by_type fp ON (p.num_followers,fp.score_type) ORDER BY score DESC LIMIT 10) SELECT fid.publication_id AS publication_id, fid.publication_id AS profile_id FROM filtered_ids_for_comment fid WHERE fid.source_type = 'publication' JOIN comment_like_scores ccl ON fid.publication_id = ccl.id
