SELECT DISTINCT
  p.id AS profile_id,
  p.username,
  p.full_name
FROM profile p
JOIN publication pub ON p.id = pub.profile_id
WHERE
  p.biography_lexemes @@ to_tsquery('english', 'italian & restaurant')
  AND pub.description_lexemes @@ to_tsquery('english', 'menu')
  AND pub.published_at >= '2025-01-01 00:00:00'
  AND pub.published_at < '2026-01-01 00:00:00'
ORDER BY
  ts_rank(p.biography_lexemes, to_tsquery('english', 'italian & restaurant')) DESC,
  pub.published_at DESC
LIMIT 10;
