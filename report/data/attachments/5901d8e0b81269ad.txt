SELECT 
    p.id AS profile_id, 
    p.username, 
    p.num_followers, 
    p.num_medias 
FROM 
    profile p 
WHERE 
    p.is_brand = TRUE 
    AND p.biography_lexemes @@ to_tsquery('english', 'restaurant | food | dining')
ORDER BY 
    (p.num_followers / NULLIF(EXTRACT(EPOCH FROM (NOW() - p.visited_at)) / 86400, 0)) DESC
LIMIT 10
