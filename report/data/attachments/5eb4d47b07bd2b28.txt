SELECT p.id AS profile_id, 111 * DEGREES(ACOS(SIN(l.lat * pi() / 180) * SIN(p.geometry::geography->'coordinates'::integer[]->0 * pi() / 180) + COS(l.lat * pi() / 180) * COS(p.geometry::geography->'coordinates'::integer[]->0 * pi() / 180) * SIN(l.lng * pi() / 180 - p.geometry::geography->'coordinates'::integer[]->1 * pi() / 180))) AS distance FROM public.profile p JOIN (SELECT id, lat, lng FROM public.profile WHERE username ILIKE '%Balne%Cambori%u%' LIMIT 10) AS l ON TRUE ORDER BY distance ASC LIMIT 10
