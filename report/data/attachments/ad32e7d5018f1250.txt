SELECT ip.id AS profile_id, p.id AS publication_id FROM public.profile_relationship AS pr JOIN public.publication AS p ON pr.destination_profile_id = p.id WHERE pr.edge = 'follows' AND p.id IN ( SELECT pp.profile_id FROM public.publication_collab AS pc JOIN public.profile AS pp ON pc.profile_id = pp.id WHERE pp.id IN ( SELECT publication_id FROM ( SELECT publication_id, ts_rank(description_lexemes, to_tsquery('english', 'restaurant')) - ts_rank description_lexemes) FROM public.publication ORDER BY 1 DESC LIMIT 10 ) AND p.published_at >= NOW() - INTERVAL '3 month' ORDER BY length(p.description_lexemes) + length(ip.biography_lexemes), ts_rank((p.description_lexemes || ip.biography_lexemes)::tsquery, to_tsquery('english', 'restaurant')) DESC LIMIT 10
