WITH follow_counts AS (SELECT profile_id, SUM(CASE WHEN edge = 'follows' THEN -1 ELSE 1 END) AS num_followers FROM profile_relationship GROUP BY origin_profile_id), engagement_scores AS (SELECT p.id AS profile_id, pub.id AS publication_id, (pub.num_likes + COALESCE(fc.num_followers, 0)) AS engagement FROM profile AS p INNER JOIN follow_counts AS fc ON fc.profile_id = p.id INNER JOIN publication AS pub ON pub.profile_id = p.id INNER JOIN profile_relationship AS er ON pub.profile_id = er.destination_profile_id WHERE p.username='@haight_clothing' GROUP BY p.id, pub.id) SELECT DISTINCT profile_id, publication_id FROM engagement_scores ORDER BY engagement DESC LIMIT 1
