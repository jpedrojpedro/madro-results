SELECT
  p.id AS profile_id,
  p.username,
  p.full_name,
  p.num_followers
FROM profile p
WHERE
  (p.biography_lexemes @@ to_tsquery('portuguese', 'restaurante | gastronomia | culinária'))
  AND (p.biography ILIKE '%São Paulo%' OR p.biography ILIKE '%SP%' OR p.full_name ILIKE '%São Paulo%' OR p.full_name ILIKE '%SP%')
  AND p.num_followers > 10000
ORDER BY p.num_followers DESC
LIMIT 10;
