WITH last_matches AS (SELECT p.profile_id AS profile_id, MAX(p.published_at) AS date FROM public.publication p JOIN public.profile r ON p.profile_id = r.id WHERE r.username = '@fluminensefc' GROUP BY p.profile_id) SELECT lm.profile_id, COUNT(*) as match_count FROM last_matches lm INNER JOIN public.comment c ON lm.date = (SELECT published_at FROM public.publication p2 JOIN public.comment cm ON p2.id = cm.pubication_id ORDER BY published_at DESC LIMIT 1)
