SELECT
  p.id AS profile_id,
  p.username,
  p.full_name
FROM profile p
WHERE
  p.biography_lexemes @@ to_tsquery('english', 'italian & (pancetta | prosciutto)')
  OR EXISTS (
    SELECT 1
    FROM publication pub
    WHERE
      pub.profile_id = p.id
      AND pub.description_lexemes @@ to_tsquery('english', 'italian & (pancetta | prosciutto)')
  )
ORDER BY
  p.num_followers DESC
LIMIT 10
