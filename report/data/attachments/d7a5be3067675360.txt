WITH associated_pubs AS (SELECT p.id AS profile_id, pub.id AS publication_id FROM profile p JOIN publication_collab pc ON p.id = pc.profile_id JOIN publication pub ON pub.id = pc.publication_id WHERE pub.slug ILIKE '%fluminense%'), comments AS (SELECT c.profile_id FROM comment c JOIN publication pub ON pub.id = c.publication_id WHERE pub.slug ILIKE 'fluminense-fc' OR pub.slug ILIKE '%fluminense%') SELECT c.profile_id, ts_rank(pub.description_lexemes, to_tsquery('english', 'most beloved player')) AS score FROM comment c JOIN associated_pubs asubs ON c.publication_id = sub.publication_id JOIN publication pub ON pub.id = sub.publication_id WHERE c.profile_id IN (SELECT profile_idsFROM comments) ORDER BY score DESC LIMIT 10
