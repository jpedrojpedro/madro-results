WITH ranked_publications AS (SELECT rp.id AS publication_id, r.id AS profile_id FROM publication rp JOIN publication_collab pc ON rp.id = pc.publication_id JOIN profile r ON pc.profile_id = r.id WHERE rp.description_lexemes @@ to_tsquery('english', 'best restaurant') OR r.biography_lexemes @@ to tsquery('english', 'best restaurant')) ORDER BY ts_rank,rp.description_lexemes, to_tsquery('english', 'best restaurant'), NULL) + ts_rank(r.biography_lexemes, to_tsquery('english', 'best restaurant')) * 0.5 DESC LIMIT 10}
