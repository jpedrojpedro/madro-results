WITH publication_lexemes AS (  SELECT publication.id AS publication_id, ts_rank(publication.description_lexemes, to_tsquery('english', 'pasta')) AS relevance FROM publication JOIN public.profile ON publication.profile_id = profile.id WHERE NOT public.is_private AND profile.name LIKE '%Bale%rio Camboriu%' ) SELECT DISTINCT publication_id ,  publication_id FROM publication_lexemes ORDER BY relevance DESC LIMIT 10;
