SELECT
  p.id AS profile_id
FROM profile p
WHERE p.biography_lexemes @@ to_tsquery('english', 'italian & (restaurant | trattoria | pasta | pizza)')
  OR p.biography_lexemes @@ to_tsquery('english', 'pancetta | prosciutto')
ORDER BY ts_rank(p.biography_lexemes, to_tsquery('english', 'italian & (restaurant | trattoria | pasta | pizza) | pancetta | prosciutto')) DESC
LIMIT 10;
