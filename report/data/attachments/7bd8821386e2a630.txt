WITH ranked_publications AS ( SELECT pub.id AS publication_id, ts_rank(pub.description_lexemes, to_tsquery('english', 'restaurant or menu or dish')) AS rank, pub.published_at AS published_at FROM public.publication AS pub WHERE description_lexemes @@ to_tsquery('english', 'restaurant or menu or dish') ORDER BY rank DESC, published_at DESC LIMIT 10 ) SELECT publication_id FROM ranked_publications;
