SELECT
  p.id AS publication_id
FROM publication p
JOIN profile pr ON p.profile_id = pr.id
WHERE
  (pr.biography_lexemes @@ to_tsquery('portuguese', 'restaurante | restaurante') AND
   pr.biography_lexemes @@ to_tsquery('portuguese', 'balneario & camboriu'))
  AND p.description_lexemes @@ to_tsquery('portuguese', 'pasta')
ORDER BY p.published_at DESC
LIMIT 10
