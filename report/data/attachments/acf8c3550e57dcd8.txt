WITH target_profile AS (
    SELECT id 
    FROM profile 
    WHERE username = 'fluminensefc'
),
fluminense_posts AS (
    SELECT id 
    FROM publication 
    WHERE profile_id = (SELECT id FROM target_profile)
),
player_interactions AS (
    SELECT c.profile_id, SUM(c.num_likes) as total_likes_received
    FROM comment c
    JOIN fluminense_posts fp ON c.publication_id = fp.id
    WHERE c.profile_id != (SELECT id FROM target_profile)
    GROUP BY c.profile_id
)
SELECT p.id AS profile_id
FROM profile p
JOIN player_interactions pi ON p.id = pi.profile_id
ORDER BY pi.total_likes_received DESC NULLS LAST
LIMIT 10
