SELECT 
    p.id AS profile_id
FROM profile p
WHERE p.biography_lexemes @@ to_tsquery('english', 'bakery')
AND EXISTS (
    SELECT 1 
    FROM profile_relationship pr
    JOIN profile influencer ON pr.origin_profile_id = influencer.id
    WHERE pr.destination_profile_id = p.id
    AND influencer.num_followers > 10000
)
ORDER BY p.num_followers DESC
LIMIT 10;
