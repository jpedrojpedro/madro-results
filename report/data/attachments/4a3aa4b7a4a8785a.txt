SELECT 
    p.id AS profile_id,
    p.full_name,
    p.username,
    AVG(c.num_likes) AS avg_comment_likes
FROM profile p
JOIN publication pub ON p.id = pub.profile_id
JOIN comment c ON pub.id = c.publication_id
WHERE p.biography_lexemes @@ to_tsquery('english', 'restaurant | food | eatery | cafe | dining')
GROUP BY p.id, p.full_name, p.username
ORDER BY avg_comment_likes DESC NULLS LAST
LIMIT 10
