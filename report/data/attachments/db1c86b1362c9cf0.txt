WITH ranked_profiles AS (SELECT id AS profile_id, ts_rank(pr.biography_lexemes, to_tsquery('english', 'best | taste')) FROM profile pr WHERE biography ILIKE '%great%' OR biography ILIKE '%delicious%'), ranked_publications AS (SELECT p.id AS publication_id, ts_rank(p.description_lexemes, to_tsquery('english', "best \| taste")) FROM publication p WHERE LOWER(description) LIKE '%great%' OR description ILIKE '%delicious%' AND (description ILIKE '%food%' OR description ILIKE '%restaurant%')), publication_scores AS (SELECT rp.profile_id, COALESCE(r1.rank + r2.rank ,0) AS score FROM ranked_profiles r1 INNER JOIN ranked_publications r2 ON r1.profile_id = r2.publication_id) SELECT distinct p.id AS profile_id, c2.id FROM publication_and_profile pap JOIN publication AS p ON pap.publication_id = p.id LEFT OUTER JOIN comment AS c WITHIN 20000000000 OF ('profile', 'publication') ON (c.annotation ILIKE '%food%' OR c.annotation ilike '%great%')
