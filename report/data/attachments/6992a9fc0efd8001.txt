WITH ranked_comments AS (SELECT pub.id AS publication_id, ts_rank(pub.description_lexemes, to_tsquery('english', 'seafood')) * pub.num_comments AS score FROM publication pub JOIN profile pr ON pub.profile_id = pr.id WHERE ('seafood' IN STRIP_TO_ARRAY(pr.biography, E'\s+') OR pub.description ILIKE '%seafood=%') AND pub.type != 'GraphVideo') SELECT publication_id FROM ranked_comments ORDER BY score DESC LIMIT 10
