SELECT p.id AS profile_id, pub.id AS publication_id FROM profile p JOIN publication pub ON p.id = pub.profile_id WHERE (pub.description_lexemes @@ to_tsquery('english', 'pasta')) AND EXISTS ( SELECT 1 FROM publication_collab pc WHERE pc.publication_id = pub.id AND pc.profile_id IN (SELECT id FROM profile WHERE full_name LIKE '%Balneário Camboriú%' )) ORDER BY ts_rank(pub.description_lexemes, to_tsquery('english', 'pasta')) LIMIT 10
